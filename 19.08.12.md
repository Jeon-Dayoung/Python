## **List**
```python
def test_list():
    scores = [10, 20, 30]
    assert scores[0] == 10
    assert scores[1] == 20
    assert scores[2] == 30
    assert scores[-1] == 30
    assert scores[-2] == 20
    assert scores[:2] == [10, 20]
    assert scores[1:] == [20, 30]

# list에 5를 추가해보자
    scores.append(5)
    assert scores == [10, 20, 30, 5]

 # 바꾸기
    scores[0] = 0
    assert scores == [0, 20, 30, 5]
```
## **Dectionary**
```python
def test_dictionary():
    scores = {'국어': 10, '영어': 20, '수학': 30}
    assert scores['국어'] == 10
    assert scores['영어'] == 20
    assert scores['수학'] == 30
    
     # 바꾸기
    scores = {'국어': 100}
    assert scores['국어'] == 100
```
```python
def test_dictionary2():
```
